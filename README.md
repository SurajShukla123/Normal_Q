1. Question

12345

12
1

Solution

class Codechef
{
	public static void main (String[] args) 
	{
	
	int row, col,n=5;
	
	for (row=n;row>=1;row--)
	{
	    for(col=1;col<=row;col++)
	    {
	      
	        System.out.print(col);
	    }
	    System.out.println();
	}
	
	}
}
